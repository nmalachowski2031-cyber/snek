document.addEventListener("DOMContentLoaded", function(){ const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); const scoreDisplay = document.getElementById("score"); const colorPicker = document.getElementById("snakeColor"); const difficulty = document.getElementById("difficulty"); const box = 20; const size = 400; let snake, snake2, food, score, mode, running = false; let direction, nextDirection; let direction2, nextDirection2; // Animation timing variables let lastTime = 0; let moveAccumulator = 0; // Button Listeners document.getElementById("normal").onclick = () => startGame("normal"); document.getElementById("endless").onclick = () => startGame("endless"); document.getElementById("roulette").onclick = () => startGame("roulette"); document.getElementById("multi").onclick = () => startGame("multiplayer"); document.getElementById("restart").onclick = () => startGame(mode); document.getElementById("menuBtn").onclick = backToMenu; function startGame(selectedMode){ mode = selectedMode; score = 0; moveAccumulator = 0; running = true; // Initial setup - P1 starts with two segments to allow smooth sliding snake = [{x: 200, y: 200}, {x: 180, y: 200}]; direction = nextDirection = "RIGHT"; if(mode === "multiplayer"){ snake2 = [{x: 100, y: 100}, {x: 80, y: 100}]; direction2 = nextDirection2 = "RIGHT"; } food = spawnFood(); scoreDisplay.innerText = score; showScreen("gameScreen"); // Start the high-fps loop requestAnimationFrame(gameLoop); } function spawnFood(){ return { x: Math.floor(Math.random() * (size / box)) * box, y: Math.floor(Math.random() * (size / box)) * box }; } document.addEventListener("keydown", e => { const key = e.key; if(key === "ArrowLeft" && direction !== "RIGHT") nextDirection = "LEFT"; if(key === "ArrowRight" && direction !== "LEFT") nextDirection = "RIGHT"; if(key === "ArrowUp" && direction !== "DOWN") nextDirection = "UP"; if(key === "ArrowDown" && direction !== "UP") nextDirection = "DOWN"; if(mode === "multiplayer"){ const k = key.toLowerCase(); if(k === "a" && direction2 !== "RIGHT") nextDirection2 = "LEFT"; if(k === "d" && direction2 !== "LEFT") nextDirection2 = "RIGHT"; if(k === "w" && direction2 !== "DOWN") nextDirection2 = "UP"; if(k === "s" && direction2 !== "UP") nextDirection2 = "DOWN"; } }); function getNextHead(currentHead, dir) { let n = { x: currentHead.x, y: currentHead.y }; if(dir === "LEFT") n.x -= box; if(dir === "RIGHT") n.x += box; if(dir === "UP") n.y -= box; if(dir === "DOWN") n.y += box; return n; } function gameLoop(timestamp) { if(!running) return; const deltaTime = timestamp - lastTime; lastTime = timestamp; // How long it takes to move one full square (difficulty) const moveInterval = parseInt(difficulty.value); moveAccumulator += deltaTime; // When accumulator reaches the interval, update logic positions if (moveAccumulator >= moveInterval) { moveAccumulator = 0; updateLogic(); } // Draw every frame using interpolation (progress between 0 and 1) draw(moveAccumulator / moveInterval); requestAnimationFrame(gameLoop); } function updateLogic() { direction = nextDirection; let head1 = getNextHead(snake[0], direction); // Border and Collision if(mode === "endless") { if(head1.x < 0) head1.x = size - box; else if(head1.x >= size) head1.x = 0; if(head1.y < 0) head1.y = size - box; else if(head1.y >= size) head1.y = 0; } else if (head1.x < 0 || head1.y < 0 || head1.x >= size || head1.y >= size) return gameOver(); if(snake.some(seg => seg.x === head1.x && seg.y === head1.y)) return gameOver(); snake.unshift(head1); if(head1.x === food.x && head1.y === food.y){ score++; scoreDisplay.innerText = score; food = spawnFood(); } else snake.pop(); if(mode === "multiplayer"){ direction2 = nextDirection2; let head2 = getNextHead(snake2[0], direction2); if(head2.x < 0 || head2.y < 0 || head2.x >= size || head2.y >= size || snake2.some(seg => seg.x === head2.x && seg.y === head2.y)) return gameOver(); snake2.unshift(head2); if(head2.x === food.x && head2.y === food.y) food = spawnFood(); else snake2.pop(); } } function draw(interpolation) { ctx.fillStyle = "black"; ctx.fillRect(0, 0, size, size); // Draw Food ctx.fillStyle = "red"; ctx.fillRect(food.x, food.y, box, box); // Draw Snake 1 with smoothness ctx.fillStyle = colorPicker.value || "#00FF00"; // Fallback to green if picker fails renderSmoothSnake(snake, direction, interpolation); if(mode === "multiplayer") { ctx.fillStyle = "#00d4ff"; renderSmoothSnake(snake2, direction2, interpolation); } } function renderSmoothSnake(body, dir, progress) { body.forEach((seg, i) => { // Segment i slides toward the position previously held by segment i-1 let target = (i === 0) ? getNextHead(seg, dir) : body[i-1]; let renderX = seg.x + (target.x - seg.x) * progress; let renderY = seg.y + (target.y - seg.y) * progress; // Prevent weird sliding lines when wrapping through walls in Endless Mode if (Math.abs(target.x - seg.x) > box) renderX = seg.x; if (Math.abs(target.y - seg.y) > box) renderY = seg.y; ctx.fillRect(renderX + 1, renderY + 1, box - 2, box - 2); }); } function showScreen(id){ document.querySelectorAll('.screen').forEach(s => s.className="screen"); document.getElementById(id).className="screen active"; } function gameOver(){ running = false; document.getElementById("finalScore").innerText = "Final Score: " + score; showScreen("gameOverScreen"); } function backToMenu(){ running = false; showScreen("menu"); } });
