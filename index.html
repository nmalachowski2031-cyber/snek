document.addEventListener("DOMContentLoaded", function(){ const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); const scoreDisplay = document.getElementById("score"); const colorPicker = document.getElementById("snakeColor"); const difficulty = document.getElementById("difficulty"); const box = 20; const size = 400; // Core Game State let snake, snake2, food, score, mode, running = false; let direction, nextDirection; let direction2, nextDirection2; // Smooth Movement Variables let lastTime = 0; let moveProgress = 0; // 0 to 1 (percentage between cells) // Listeners document.getElementById("normal").onclick = () => startGame("normal"); document.getElementById("endless").onclick = () => startGame("endless"); document.getElementById("roulette").onclick = () => startGame("roulette"); document.getElementById("multi").onclick = () => startGame("multiplayer"); document.getElementById("restart").onclick = () => startGame(mode); document.getElementById("menuBtn").onclick = backToMenu; function startGame(selectedMode){ mode = selectedMode; score = 0; moveProgress = 0; running = true; // Initial setup snake = [{x: 200, y: 200}, {x: 180, y: 200}]; direction = nextDirection = "RIGHT"; if(mode === "multiplayer"){ snake2 = [{x: 100, y: 100}, {x: 80, y: 100}]; direction2 = nextDirection2 = "RIGHT"; } food = spawnFood(); scoreDisplay.innerText = score; document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById("gameScreen").classList.add('active'); requestAnimationFrame(gameLoop); } function spawnFood(){ return { x: Math.floor(Math.random() * (size / box)) * box, y: Math.floor(Math.random() * (size / box)) * box }; } document.addEventListener("keydown", e => { const key = e.key; if(key === "ArrowLeft" && direction !== "RIGHT") nextDirection = "LEFT"; if(key === "ArrowRight" && direction !== "LEFT") nextDirection = "RIGHT"; if(key === "ArrowUp" && direction !== "DOWN") nextDirection = "UP"; if(key === "ArrowDown" && direction !== "UP") nextDirection = "DOWN"; if(mode === "multiplayer"){ const k = key.toLowerCase(); if(k === "a" && direction2 !== "RIGHT") nextDirection2 = "LEFT"; if(k === "d" && direction2 !== "LEFT") nextDirection2 = "RIGHT"; if(k === "w" && direction2 !== "DOWN") nextDirection2 = "UP"; if(k === "s" && direction2 !== "UP") nextDirection2 = "DOWN"; } }); function getNextHead(head, dir) { let n = { x: head.x, y: head.y }; if(dir === "LEFT") n.x -= box; if(dir === "RIGHT") n.x += box; if(dir === "UP") n.y -= box; if(dir === "DOWN") n.y += box; return n; } function gameLoop(timestamp) { if(!running) return; const deltaTime = timestamp - lastTime; lastTime = timestamp; // Calculate speed based on difficulty (milliseconds per cell) const moveSpeed = parseInt(difficulty.value); moveProgress += deltaTime / moveSpeed; if (moveProgress >= 1) { moveProgress = 0; updateLogic(); } draw(moveProgress); requestAnimationFrame(gameLoop); } function updateLogic() { direction = nextDirection; let head1 = getNextHead(snake[0], direction); // Border Logic if(mode === "endless") { if(head1.x < 0) head1.x = size - box; else if(head1.x >= size) head1.x = 0; if(head1.y < 0) head1.y = size - box; else if(head1.y >= size) head1.y = 0; } else if (head1.x < 0 || head1.y < 0 || head1.x >= size || head1.y >= size) return gameOver(); // Self Collision if(snake.some(seg => seg.x === head1.x && seg.y === head1.y)) return gameOver(); snake.unshift(head1); if(head1.x === food.x && head1.y === food.y){ score++; scoreDisplay.innerText = score; food = spawnFood(); } else snake.pop(); if(mode === "multiplayer"){ direction2 = nextDirection2; let head2 = getNextHead(snake2[0], direction2); if(head2.x < 0 || head2.y < 0 || head2.x >= size || head2.y >= size || snake2.some(seg => seg.x === head2.x && seg.y === head2.y)) return gameOver(); snake2.unshift(head2); if(head2.x === food.x && head2.y === food.y) food = spawnFood(); else snake2.pop(); } } function draw(progress) { ctx.fillStyle = "black"; ctx.fillRect(0, 0, size, size); // Draw Food ctx.fillStyle = "red"; ctx.fillRect(food.x, food.y, box, box); // Render Snake 1 with Interpolation ctx.fillStyle = colorPicker.value; // FIXED: Always uses current picker value drawSnakeSmooth(snake, direction, progress); if(mode === "multiplayer") { ctx.fillStyle = "#00d4ff"; drawSnakeSmooth(snake2, direction2, progress); } } function drawSnakeSmooth(sArr, dir, progress) { sArr.forEach((seg, i) => { let nextPos = (i === 0) ? getNextHead(seg, dir) : sArr[i-1]; // Seamlessly slide between current and next position let renderX = seg.x + (nextPos.x - seg.x) * progress; let renderY = seg.y + (nextPos.y - seg.y) * progress; // Handle wrap-around visual jumps in endless mode if (Math.abs(nextPos.x - seg.x) > box) renderX = seg.x; if (Math.abs(nextPos.y - seg.y) > box) renderY = seg.y; ctx.fillRect(renderX + 1, renderY + 1, box - 2, box - 2); }); } function gameOver(){ running = false; document.getElementById("finalScore").innerText = "Final Score: " + score; document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById("gameOverScreen").classList.add('active'); } function backToMenu(){ running = false; document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById("menu").classList.add('active'); } });
